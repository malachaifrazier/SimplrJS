<script type="text/javascript">
	QUnit.config.reorder = false
	$(function() {
		
		module("Simplr.Conversion");
		test("Simplr.Conversion.mObjectToJSONString()", function() {
			same(Simplr.Conversion.mObjectToJSONString({ a : 1, b : 2 }), '{"a":1,"b":2}', "Object -> String");
			same(Simplr.Conversion.mObjectToJSONString({ a : { x : "1", y : "abc" }, b : 2 }), '{"a":{"x":"1","y":"abc"},"b":2}', "Object -> String");
			same(Simplr.Conversion.mObjectToJSONString([ 1, "2", 3, "4" ]), '[1,"2",3,"4"]', "Object -> String");
		});

		test("Simplr.Conversion.mJSONStringToObject()", function() {
			same(Simplr.Conversion.mJSONStringToObject('{"a":1,"b":2}'), { a : 1, b : 2 }, "String -> Object");
			same(Simplr.Conversion.mJSONStringToObject('{"a":{"x":"1","y":"abc"},"b":2}'), { a : { x : "1", y : "abc" }, b : 2 } , "String -> Object");
			same(Simplr.Conversion.mJSONStringToObject('[1,"2",3,"4"]'), [ 1, "2", 3, "4" ], "String -> Object");
		});
	
	});
</script>
